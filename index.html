Tata Bahasa (文法)

`; data.bunpou.forEach(item => { html += `

${item.pola}

${item.penjelasan}

${item.jp}

${item.id}

`; }); document.getElementById('content').innerHTML = html; } function setupNav() { const nextBtn = document.getElementById('next-btn'); const prevBtn = document.getElementById('prev-btn'); if (currentBab < 50) nextBtn.onclick = () => window.location.href = `template.html?bab=${currentBab + 1}`; else { nextBtn.innerText = "🎉 Selesai!"; nextBtn.onclick = () => window.location.href = 'index.html'; } if (currentBab > 1) { prevBtn.style.display = "block"; prevBtn.onclick = () => window.location.href = `template.html?bab=${currentBab - 1}`; } }
